def enConvertor enNumbers
  persianNumbers = %w(. ۱ ۲ ۳ ۴ ۵ ۶ ۷ ۸ ۹)
  enNumbers.to_s.chars.map{|en| persianNumbers[en.to_i]}.join()
end

  def toPersian num

    num = num.chars.map{|char| ARABIC_NUMBERS[char.to_i]}.join
